package com.sathishn.portfolio.controller;

@RestController
@RequestMapping("/api/v1/profile")
public class SathishController {

    @Autowired
    private DeveloperService devService;

    @GetMapping("/about")
    public ResponseEntity<Profile> getAboutInfo() {
        Profile profile = devService.getSathishProfile();
        return ResponseEntity.ok()
            .header("X-Developer-Status", "ACTIVE")
            .body(profile);
    }

    @PostMapping("/hire")
    public ResponseEntity<String> contact(@RequestBody Message msg) {
        devService.dispatchContactForm(msg);
        return ResponseEntity.status(201)
            .body("Message received. Action initialized!");
    }
}
